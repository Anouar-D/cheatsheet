git cheatsheet

# branch aanmaken
- git checkout -b { branch-name }

# git aanmaken
- git init
- git commit -m "initial"
- git add .
- git remote add origin { https/SSH }
- git push --set-upstream origin { branch }

# git pushen
- git add .
- git remote add origin { https/SHH }
- git push --set-upstrean origin { branch }
